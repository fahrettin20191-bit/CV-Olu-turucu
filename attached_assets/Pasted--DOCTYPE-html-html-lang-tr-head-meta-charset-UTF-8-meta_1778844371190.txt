    <script>
        let experienceCount = 0;
        let educationCount = 0;
        
        // Fotoğraf yükleme
        document.getElementById('photoInput').addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    document.getElementById('photoPreview').innerHTML = 
                        `<img src="${e.target.result}" style="width:100%; height:100%; object-fit: cover;">`;
                    document.getElementById('previewPhoto').innerHTML = 
                        `<img src="${e.target.result}" style="width:100%; height:100%; object-fit: cover;">`;
                };
                reader.readAsDataURL(file);
            }
        });
        
        // İş deneyimi ekle
        function addExperience() {
            experienceCount++;
            const div = document.createElement('div');
            div.className = 'experience-item';
            div.id = `exp-${experienceCount}`;
            div.innerHTML = `
                <button class="remove-btn" onclick="removeExperience(${experienceCount})">Kaldır</button>
                <input type="text" placeholder="Şirket Adı" class="exp-company" data-type="company">
                <input type="text" placeholder="Pozisyon" class="exp-position" data-type="position">
                <input type="text" placeholder="Tarih (Ör: 2020 - Halen)" class="exp-date" data-type="date">
                <textarea placeholder="Açıklama" class="exp-description" data-type="description"></textarea>
            `;
            document.getElementById('experiences').appendChild(div);
        }
        
        // Eğitim ekle
        function addEducation() {
            educationCount++;
            const div = document.createElement('div');
            div.className = 'education-item';
            div.id = `edu-${educationCount}`;
            div.innerHTML = `
                <button class="remove-btn" onclick="removeEducation(${educationCount})">Kaldır</button>
                <input type="text" placeholder="Üniversite / Okul" class="edu-school" data-type="school">
                <input type="text" placeholder="Bölüm / Derece" class="edu-degree" data-type="degree">
                <input type="text" placeholder="Tarih (Ör: 2016 - 2020)" class="edu-date" data-type="date">
            `;
            document.getElementById('educations').appendChild(div);
        }
        
        function removeExperience(id) {
            document.getElementById(`exp-${id}`).remove();
        }
        
        function removeEducation(id) {
            document.getElementById(`edu-${id}`).remove();
        }
        
        function updatePreview() {
            // Kişisel bilgiler
            document.getElementById('previewName').textContent = document.getElementById('name').value || 'Ad Soyad';
            document.getElementById('previewTitle').textContent = document.getElementById('title').value || 'Unvan';
            
            // İletişim
            let contact = '';
            const phone = document.getElementById('phone').value;
            const email = document.getElementById('email').value;
            const linkedin = document.getElementById('linkedin').value;
            
            if (phone) contact += `Telefon: ${phone}<br>`;
            if (email) contact += `E-posta: ${email}<br>`;
            if (linkedin) contact += `LinkedIn: ${linkedin}<br>`;
            
            document.getElementById('previewContact').innerHTML = contact || 'İletişim bilgileri ekleyin';
            
            // Yetenekler
            const skills = document.getElementById('skills').value;
            document.getElementById('previewSkills').innerHTML = skills ? 
                skills.split(',').map(s => s.trim()).join(' • ') : 'Yeteneklerinizi ekleyin';
            
            // İş deneyimi
            let expHtml = '';
            document.querySelectorAll('.experience-item').forEach(item => {
                const company = item.querySelector('[data-type="company"]').value;
                const position = item.querySelector('[data-type="position"]').value;
                const date = item.querySelector('[data-type="date"]').value;
                const desc = item.querySelector('[data-type="description"]').value;
                
                if (company || position) {
                    expHtml += `
                        <div class="cv-item">
                            <div class="cv-company">${company || 'Şirket'}</div>
                            <div>${position || 'Pozisyon'} ${date ? `| ${date}` : ''}</div>
                            ${desc ? `<div>${desc}</div>` : ''}
                        </div>
                    `;
                }
            });
            document.getElementById('previewExperience').innerHTML = expHtml || 'İş deneyiminizi ekleyin';
            
            // Eğitim
            let eduHtml = '';
            document.querySelectorAll('.education-item').forEach(item => {
                const school = item.querySelector('[data-type="school"]').value;
                const degree = item.querySelector('[data-type="degree"]').value;
                const date = item.querySelector('[data-type="date"]').value;
                
                if (school || degree) {
                    eduHtml += `
                        <div class="cv-item">
                            <div class="cv-company">${school || 'Okul'}</div>
                            <div>${degree || 'Derece'} ${date ? `| ${date}` : ''}</div>
                        </div>
                    `;
                }
            });
            document.getElementById('previewEducation').innerHTML = eduHtml || 'Eğitim bilgilerinizi ekleyin';
        }
        
        async function downloadPDF() {
            const loading = document.getElementById('loading');
            const preview = document.getElementById('cvPreview');
            
            loading.style.display = 'block';
            
            try {
                const canvas = await html2canvas(preview, {
                    scale: 2,
                    useCORS: true,
                    backgroundColor: '#ffffff',
                    width: preview.scrollWidth,
                    height: preview.scrollHeight
                });
                
                const imgData = canvas.toDataURL('image/png');
                const { jsPDF } = window.jspdf;
                const pdf = new jsPDF('p', 'mm', 'a4');
                
                const imgWidth = 210;
                const pageHeight = 295;
                const imgHeight = (canvas.height * imgWidth) / canvas.width;
                let heightLeft = imgHeight;
                let position = 0;
                
                pdf.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);
                heightLeft -= pageHeight;
                
                while (heightLeft >= 0) {
                    position = heightLeft - imgHeight;
                    pdf.addPage();
                    pdf.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);
                    heightLeft -= pageHeight;
                }
                
                const name = document.getElementById('name').value.replace(/\s+/g, '_') || 'CV';
                pdf.save(`${name}_CV.pdf`);
                
            } catch (error) {
                alert('PDF oluşturma hatası! Tarayıcınızı yenileyip tekrar deneyin.');
            } finally {
                loading.style.display = 'none';
            }
        }

            addEducation();
            updatePreview();
        };
